function binaryAgent(str) {
  
  var str = str.split(' ');
  var sentence = ""
  
  //convert binary to decimal, consider decimals as CharCodes and build the sentence
  for(var i = 0; i < str.length; i++) {
    str[i] = parseInt(str[i], 2)
    sentence += String.fromCharCode(str[i])
  }
  
  return sentence;
}
